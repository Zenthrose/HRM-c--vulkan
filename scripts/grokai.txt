Open the file that creates the Vulkan device (probably hrm_main.cpp or a Vulkan init function).
Find the vkCreateDevice call.
Add this right after it:
C++VkResult result = vkCreateDevice(physicalDevice, &createInfo, nullptr, &device);
if (result != VK_SUCCESS) {
    std::cerr << "vkCreateDevice failed with error: " << result << std::endl;
    switch (result) {
        case VK_ERROR_EXTENSION_NOT_PRESENT: std::cerr << "Missing required extension" << std::endl; break;
        case VK_ERROR_FEATURE_NOT_PRESENT:   std::cerr << "Missing required feature" << std::endl; break;
        case VK_ERROR_INITIALIZATION_FAILED: std::cerr << "Driver bug or bad params" << std::endl; break;
    }
    device = VK_NULL_HANDLE;
}
